// 进制转换（十进制转二进制）
void decToBin(int n) {
    if (n == 0) return;
    decToBin(n / 2);
    cout << n % 2;
}

// 位运算常用技巧
if (n & 1) // 判断奇数
n = n ^ m; // 交换（不借助临时变量，需配合）
n = n << 1; // 乘以2
n = n >> 1; // 除以2

// 枚举法（判断素数）
bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i++)
        if (n % i == 0) return false;
    return true;
}

// 一维数组基础
int a[100];
for (int i = 0; i < n; i++) cin >> a[i];

// 冒泡排序
void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - 1 - i; j++)
            if (arr[j] > arr[j+1])
                swap(arr[j], arr[j+1]);
}

// 选择排序（不稳定）
void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++)
            if (arr[j] < arr[minIdx]) minIdx = j;
        swap(arr[i], arr[minIdx]);
    }
}

// 插入排序（稳定）
void insertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i], j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j+1] = arr[j];
            j--;
        }
        arr[j+1] = key;
    }
}

int gcd(int a, int b) {
    return b == 0 ? a : gcd(b, a % b);
}

// 2. 素数判定（试除法）
bool isPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i * i <= n; i++)
        if (n % i == 0) return false;
    return true;
}

// 3. 埃氏筛法
void sieve(int n) {
    vector<bool> is_prime(n + 1, true);
    is_prime[0] = is_prime[1] = false;
    for (int i = 2; i * i <= n; i++)
        if (is_prime[i])
            for (int j = i * i; j <= n; j += i)
                is_prime[j] = false;
}

// 4. 线性筛法（欧拉筛）
const int MAXN = 1000005;
int primes[MAXN], cnt = 0;
bool is_composite[MAXN];
void linearSieve(int n) {
    for (int i = 2; i <= n; i++) {
        if (!is_composite[i]) primes[cnt++] = i;
        for (int j = 0; j < cnt && i * primes[j] <= n; j++) {
            is_composite[i * primes[j]] = true;
            if (i % primes[j] == 0) break; // 关键
        }
    }
}

// 6. 快速排序 partition
int partition(int arr[], int l, int r) {
    int pivot = arr[r];
    int i = l - 1;
    for (int j = l; j < r; j++)
        if (arr[j] < pivot) { i++; swap(arr[i], arr[j]); }
    swap(arr[i + 1], arr[r]);
    return i + 1;
}

// 7. 归并排序 merge
void merge(int arr[], int l, int m, int r) {
    int n1 = m - l + 1, n2 = r - m;
    vector<int> L(n1), R(n2);
    for (int i = 0; i < n1; i++) L[i] = arr[l + i];
    for (int j = 0; j < n2; j++) R[j] = arr[m + 1 + j];
    int i = 0, j = 0, k = l;
    while (i < n1 && j < n2)
        if (L[i] <= R[j]) arr[k++] = L[i++]; // <=保证稳定
        else arr[k++] = R[j++];
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

// 8. 二分查找（标准版）
int binarySearch(vector<int>& arr, int target) {
    int l = 0, r = arr.size() - 1;
    while (l <= r) {
        int mid = l + (r - l) / 2; // 防溢出
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) l = mid + 1;
        else r = mid - 1;
    }
    return -1;
}

// 9. 链表删除（虚拟头节点）
struct ListNode { int val; ListNode* next; ListNode(int x):val(x),next(nullptr){} };
ListNode* removeElements(ListNode* head, int val) {
    ListNode* dummy = new ListNode(0);
    dummy->next = head;
    ListNode* cur = dummy;
    while (cur->next) {
        if (cur->next->val == val) {
            ListNode* tmp = cur->next;
            cur->next = cur->next->next;
            delete tmp;
        } else cur = cur->next;
    }
    return dummy->next;
}

// 10. 高精度加法
string add(string a, string b) {
    string res;
    int i = a.size() - 1, j = b.size() - 1, carry = 0;
    while (i >= 0 || j >= 0 || carry) {
        int sum = carry;
        if (i >= 0) sum += a[i--] - '0';
        if (j >= 0) sum += b[j--] - '0';
        res.push_back(sum % 10 + '0');
        carry = sum / 10;
    }
    reverse(res.begin(), res.end());
    return res;
}

const int MOD = 1e9 + 7;
long long C[1005][1005];
void initComb(int n) {
    for (int i = 0; i <= n; i++) {
        C[i][0] = C[i][i] = 1;
        for (int j = 1; j < i; j++)
            C[i][j] = (C[i-1][j-1] + C[i-1][j]) % MOD;
    }
}

// Kruskal 最小生成树
struct Edge { int u, v, w; };
bool cmp(Edge a, Edge b) { return a.w < b.w; }
int fa[1005];
int find(int x) { return fa[x] == x ? x : fa[x] = find(fa[x]); }
void kruskal(vector<Edge>& edges, int n) {
    sort(edges.begin(), edges.end(), cmp);
    for (int i = 1; i <= n; i++) fa[i] = i;
    int cnt = 0, mst = 0;
    for (auto& e : edges) {
        int fu = find(e.u), fv = find(e.v);
        if (fu != fv) {
            fa[fu] = fv;
            mst += e.w;
            if (++cnt == n - 1) break;
        }
    }
}

// Dijkstra 单源最短路径
const int INF = 0x3f3f3f3f;
int dist[1005];
void dijkstra(int start, vector<pair<int,int>> adj[], int n) {
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>>> pq;
    fill(dist, dist + n + 1, INF);
    dist[start] = 0;
    pq.push({0, start});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : adj[u]) {
            if (dist[v] > dist[u] + w) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}

// Floyd 全源最短路径
void floyd(int n, long long dist[][105]) {
    for (int k = 1; k <= n; k++)
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
}

// 快速幂（倍增思想）
long long qpow(long long a, long long b, long long mod) {
    long long res = 1 % mod;
    while (b) {
        if (b & 1) res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}

// 等差数列求和公式优化
// 暴力：for循环累加，O(N)
// 优化：S = n * (a1 + an) / 2，O(1)
long long arithmeticSum(long long a1, long long an, long long n) {
    return n * (a1 + an) / 2;
}

- **动态规划基础概念**（9/9）：递推vs递归实现、状态转移方程、重叠子问题；"递归实现总不低于递推"是错的
- **平面几何公式**（9/9）：三角形面积（海伦公式、两边夹角`a*b*sin(theta)/2`、直角边`a/2.0*b`）、扇形面积/周长、梯形面积、勾股定理、弦长公式

#### 高频知识点（7–8/9套）
- **排列组合-有限制排列/圆桌排列**（8/9）：A不能排队首、圆桌家庭相邻就座、自编车牌限制
- **排列组合-相同元素排列**（8/9）：袋中相同球不放回取完的序列数；放回与不放回的区别
- **杨辉三角/二项式系数**（8/9）：程序填空`a[j]+=a[j-1]`；第n行和为`2^n`；与组合数`C(n,k)`的关系；二项展开式系数
- **Prim算法**（7/9）：填空`graph[u][v] >= 0 && key[v] > graph[u][v]`；时间复杂度`O(v²)`
- **Kruskal/生成树性质**（7/9）：填空`uParent != vParent`；n个顶点生成树含n-1条边；完全图生成树数量；"有多颗生成树"正确
- **图的存储**（7/9）：邻接矩阵大小`n×n`；邻接表边节点数（无向图2e/有向图e）；出边表边节点数
- **树上问题-倍增/LCA/树链剖分**（7/9）：编程题5道；`f[u][i]=f[f[u][i-1]][i-1]`模板；树上移动(DFS)、接竹竿(ST表)、树上旅行(倍增)、宝石项链(跳跃表)、大量工作沟通(树链剖分)

#### 较常考（4-6/9套）
- **OOP类与对象**（6/9）：构造/析构函数可为虚函数（构造不能、析构能）；默认构造可声明为private；派生类内存总不小于基类；纯虚函数类不能实例化；拷贝构造自动调用情况
- **数论（GCD/筛法/欧拉函数）**（6/9）：欧拉筛填空`i%primes[j]==0`break；gcd复杂度；欧拉函数性质；质数判定
- **时间复杂度分析**（6/9）：递归主体`f(n)=f(n-1)+f(n-2)+2f(n-3)`；多重循环`O(n³)`；埃氏筛复杂度；count_triple复杂度
- **LCS（最长公共子序列）**（5/9）：填空`max(dp[i-1][j],dp[i][j-1])`；滚动数组优化后**无法**还原具体序列
- **概率计算**（5/9）：条件概率（"已知有一个是女孩，另一个是男孩"）；儿女双全问题；生日问题
- **程序阅读题**（4/9）：三维DP路径计数、多重循环枚举结果、位运算组合next函数
- **二叉树遍历（BFS/DFS）复杂度**（4/9）：`O(v+e)`；判断有环/连通性

#### 偶发知识点（≤3/9套）
- **进制转换**（4/9）：七进制转三进制；二进制小数转十进制（`100.001`→4.125）
- **二维数组函数参数语法**（3/9）：`int a[][10]`合法；`int a[][]`、`int **a`非法
- **Floyd算法**（3/9）：填空`map[i][j]=map[i][k]+map[k][j]`；时间复杂度`O(v³)`
- **程序填空-核心算法**（3/9）：快速排序partition、LIS、归并排序左右区间

### 编程题趋势（18道编程题）

| 考察方向 | 出现次数 | 代表题型 |
|----------|----------|----------|
| 图论算法（最短路径/Dijkstra） | 5 | 空间跳跃(堆优化Dijkstra)、上学(多源多查询)、猫和老鼠(双源Dijkstra)、割裂(Dijkstra)、最短距离(GCD建图) |

| 组合数学/杨辉三角/组合数DP | 4 | 奖品分配(组合数递推)、手套配对(C(n,k)组合)、排队(相邻关系排列)、遍历计数(树的DFS序排列) |

**难度趋势**：
- 2023-2024年：树链剖分(NOIP提高组)、倍增ST表属于拔高内容
- 2025年：Dijkstra堆优化成为编程题标配；出现双源最短路（猫和老鼠）；最小生成树删除每条边后MST重量（次小生成树思想）
- 判断题：2025.12出现"Kruskal已选边在同一连通块必须舍弃"（正确）和"快排随机化可完全避免退化"（错误）等深度辨析题

### 高频判断题速记

| 题号 | 高频考点 | 正误规律 |
|------|----------|----------|
| 判断1 | 位运算/表达式类型 | 与7级类似；注意`char`参与运算提升为`int`；`^`是异或 |
| 判断2-3 | C++语法/排序 | 引用初始化后不可改；快排和归并平均都是`O(n log n)`；快排不稳定、归并稳定 |
| 判断4 | 排列组合 | 相同元素排列用除法去重；不相邻问题用插空法；有序选班干部用排列`A(n,k)` |
| 判断5 | 数学库 | `pow(2,5)`返回`double`32.0不是`int`；`sqrt(4)`返回`double`2.0；`log(8)`约为2.079不是3 |
| 判断6 | 图论/几何/代数 | Dijkstra朴素`O(v²)`堆优化`O(e log v)`；三角形面积公式；二次函数顶点`-b/(2a)` |
| 判断7 | 二叉树/图 | 平衡二叉树查找最坏`O(log n)`；完全图边数`n(n-1)/2`；无向图度数之和=2倍边数 |
| 判断8 | 哈希/链表 | 单链表和双向链表查找都是`O(n)`；链地址法最坏`O(n)`；开放地址法装填因子影响性能 |
| 判断9 | 图连通性/遍历 | DFS和BFS均可判断连通性；BFS更适合求无权图最短路径；判断有环可用DFS或并查集 |
| 判断10 | 组合数学/数论 | 从32人中选4人担任不同职务是排列`A(32,4)`；有性别限制时需分类讨论 |

### 超纲/考纲偏差
1. **树链剖分（重链剖分）**：2023.12"大量的工作沟通"标准解法是树链剖分+线段树/LCA，属于NOIP提高组难度
2. **ST表/倍增求LCA**：虽然考纲提到"倍增法"，但实际编程题要求完整实现LCA的`f[u][i]`倍增跳跃
3. **Dijkstra堆优化**：考纲仅提及Dijkstra算法，实际频繁要求堆优化实现（`priority_queue`）
4. **组合数取模/逆元**：编程题中多次需要计算大组合数并取模，涉及预处理阶乘和卢卡斯定理思想
5. **快速排序随机化**：2025.12判断题提及"随机化pivot可完全避免退化"是错误（只能降低概率）
